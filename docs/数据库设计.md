> 版本：v1.0

## 一、ER 关系图

```
users
  │
  ├─── 创建 ──► itineraries (行程模板)
  │                │
  │                └─── itinerary_days (每日行程)
  │                          │
  │                          └─── itinerary_nodes (节点明细)
  │
  ├─── 创建 ──► contracts (合同)
  │                │  引用 itinerary_id（快照来源）
  │                │
  │                ├─── contract_days (合同行程快照-天)
  │                │        └─── contract_nodes (合同行程快照-节点)
  │                │
  │                └─── bills (账单)
  │                      ├─── 入账 (type=income)
  │                      └─── 出账 (type=expense)
  │
  └─── 操作 ──► operation_logs (操作日志)
```

## 二、表结构详述

### 2.1 users — 用户表

```sql
CREATE TABLE users (
    id          SERIAL PRIMARY KEY,
    username    VARCHAR(50)  NOT NULL UNIQUE,
    full_name   VARCHAR(100) NOT NULL,
    password_hash VARCHAR(255) NOT NULL,          -- bcrypt hash
    role        VARCHAR(20)  NOT NULL DEFAULT 'sales',
                                                  -- admin / sales / finance / readonly
    is_active   BOOLEAN      NOT NULL DEFAULT TRUE,
    created_at  TIMESTAMP    NOT NULL DEFAULT NOW(),
    updated_at  TIMESTAMP    NOT NULL DEFAULT NOW()
);
```

---

### 2.2 itineraries — 行程模板表

```sql
CREATE TABLE itineraries (
    id              SERIAL PRIMARY KEY,
    name            VARCHAR(200) NOT NULL,         -- 行程名称，如"云南双飞7日游"
    origin          VARCHAR(100) NOT NULL,         -- 出发地
    destination     VARCHAR(200) NOT NULL,         -- 目的地（支持逗号分隔多地）
    days            SMALLINT    NOT NULL,           -- 天数
    default_pax     SMALLINT,                      -- 默认人数
    tags            TEXT[],                        -- 标签数组，如 {亲子, 蜜月}
    status          VARCHAR(20) NOT NULL DEFAULT 'draft',
                                                  -- draft / published / archived
    notes           TEXT,                          -- 备注
    created_by      INTEGER NOT NULL REFERENCES users(id),
    created_at      TIMESTAMP NOT NULL DEFAULT NOW(),
    updated_at      TIMESTAMP NOT NULL DEFAULT NOW()
);

CREATE INDEX idx_itineraries_status ON itineraries(status);
CREATE INDEX idx_itineraries_created_by ON itineraries(created_by);
```

---

### 2.3 itinerary_days — 行程每日记录

```sql
CREATE TABLE itinerary_days (
    id              SERIAL PRIMARY KEY,
    itinerary_id    INTEGER NOT NULL REFERENCES itineraries(id) ON DELETE CASCADE,
    day_number      SMALLINT NOT NULL,             -- 第几天，从 1 开始
    title           VARCHAR(200),                  -- 当天标题，如"抵达昆明"
    notes           TEXT,
    UNIQUE (itinerary_id, day_number)
);

CREATE INDEX idx_itinerary_days_itinerary ON itinerary_days(itinerary_id);
```

---

### 2.4 itinerary_nodes — 行程节点明细

```sql
CREATE TABLE itinerary_nodes (
    id              SERIAL PRIMARY KEY,
    day_id          INTEGER NOT NULL REFERENCES itinerary_days(id) ON DELETE CASCADE,
    sort_order      SMALLINT NOT NULL DEFAULT 0,   -- 排序，支持拖拽后更新
    node_type       VARCHAR(20) NOT NULL,
                    -- transport / accommodation / attraction / meal / activity / free_time
    title           VARCHAR(200) NOT NULL,          -- 节点标题
    start_time      TIME,                           -- 开始时间（可选）
    end_time        TIME,                           -- 结束时间（可选）
    duration_minutes INTEGER,                       -- 时长（分钟）
    price           NUMERIC(10, 2),                 -- 参考价格
    price_included  BOOLEAN DEFAULT TRUE,           -- 是否含在方案价格中
    extra_data      JSONB,                          -- 各类型扩展字段（见下方说明）
    notes           TEXT
);

CREATE INDEX idx_itinerary_nodes_day ON itinerary_nodes(day_id);
```

**extra_data JSONB 结构示例（按 node_type）：**

```jsonc
// transport（交通）

  "departure_time": "08:30",

// accommodation（住宿）

  "room_type": "豪华大床房",

// attraction（景点）

```

### 2.5 contracts — 合同表

```sql
CREATE TABLE contracts (
    id                  SERIAL PRIMARY KEY,
    contract_no         VARCHAR(50) NOT NULL UNIQUE,  -- 合同编号，如 TRV-2026-0001
    source_itinerary_id INTEGER REFERENCES itineraries(id) ON DELETE SET NULL,
                                                      -- 来源行程（仅记录，行程明细已快照）
    
    -- 客户信息
    customer_name       VARCHAR(100) NOT NULL,
    customer_phone      VARCHAR(20)  NOT NULL,
    customer_id_no      VARCHAR(30),                  -- 身份证号（可选）
    travelers           JSONB,                        -- 出行人员名单 [{name, id_no}, ...]
    pax                 SMALLINT     NOT NULL,         -- 出行人数
    
    -- 行程时间
    departure_date      DATE         NOT NULL,
    return_date         DATE         NOT NULL,
    
    -- 费用条款
    total_amount        NUMERIC(12, 2) NOT NULL,       -- 合同总金额
    price_per_person    NUMERIC(10, 2),
    deposit_amount      NUMERIC(10, 2),                -- 定金
    deposit_due_date    DATE,
    balance_amount      NUMERIC(10, 2),                -- 尾款（可自动计算）
    balance_due_date    DATE,
    includes            TEXT,                          -- 含什么
    excludes            TEXT,                          -- 不含什么
    cancellation_policy TEXT,                          -- 退改规则
    
    -- 状态
    status              VARCHAR(20)  NOT NULL DEFAULT 'draft',
                        -- draft / pending_sign / signed / in_progress / completed / cancelled
    cancel_reason       TEXT,
    attachment_url      TEXT,                          -- 签署扫描件
    
    notes               TEXT,
    created_by          INTEGER NOT NULL REFERENCES users(id),
    created_at          TIMESTAMP NOT NULL DEFAULT NOW(),
    updated_at          TIMESTAMP NOT NULL DEFAULT NOW()
);

CREATE INDEX idx_contracts_status ON contracts(status);
CREATE INDEX idx_contracts_customer ON contracts(customer_name);
CREATE INDEX idx_contracts_departure ON contracts(departure_date);
CREATE INDEX idx_contracts_created_by ON contracts(created_by);

-- 自动生成合同编号的序列（可在应用层生成）
CREATE SEQUENCE contract_no_seq START 1;
```

### 2.7 contract_nodes — 合同行程快照（节点）

```sql

CREATE TABLE contract_nodes (

);
```

### 2.8 bills — 账单表

```sql
CREATE TABLE bills (
    id              SERIAL PRIMARY KEY,
    contract_id     INTEGER REFERENCES contracts(id) ON DELETE SET NULL,
                                                      -- 可为空（非合同相关支出）
    bill_type       VARCHAR(10) NOT NULL,              -- income（入账）/ expense（出账）
    
    -- 入账专用
    income_type     VARCHAR(20),                       -- deposit / balance / full / other
    
    -- 出账专用
    expense_type    VARCHAR(30),
                    -- transport / accommodation / attraction / meal / guide
                    -- insurance / operation / other
    supplier_name   VARCHAR(100),
    
    -- 通用字段
    amount          NUMERIC(12, 2) NOT NULL,
    bill_date       DATE         NOT NULL,
    payment_method  VARCHAR(20)  NOT NULL,
                    -- cash / wechat / alipay / bank_transfer / pos
    account_info    VARCHAR(100),                      -- 收款账号/备注
    attachment_url  TEXT,                              -- 凭证图片（转账截图/发票）
    notes           TEXT,
    
    created_by      INTEGER NOT NULL REFERENCES users(id),
    created_at      TIMESTAMP NOT NULL DEFAULT NOW(),
    updated_at      TIMESTAMP NOT NULL DEFAULT NOW()
);

CREATE INDEX idx_bills_contract ON bills(contract_id);
CREATE INDEX idx_bills_type ON bills(bill_type);
CREATE INDEX idx_bills_date ON bills(bill_date);
CREATE INDEX idx_bills_created_by ON bills(created_by);
```

---

### 2.9 operation_logs — 操作日志

```sql
CREATE TABLE operation_logs (
    id              BIGSERIAL PRIMARY KEY,
    user_id         INTEGER REFERENCES users(id) ON DELETE SET NULL,
    action          VARCHAR(50) NOT NULL,              -- create / update / delete / status_change
    resource_type   VARCHAR(30) NOT NULL,              -- itinerary / contract / bill
    resource_id     INTEGER NOT NULL,
    detail          JSONB,                             -- 变更前后的关键字段快照
    ip_address      INET,
    created_at      TIMESTAMP NOT NULL DEFAULT NOW()
);

CREATE INDEX idx_op_logs_user ON operation_logs(user_id);
CREATE INDEX idx_op_logs_resource ON operation_logs(resource_type, resource_id);
CREATE INDEX idx_op_logs_created ON operation_logs(created_at);
```

## 三、关键设计决策

| 决策 | 说明 |
|------|------|
| **行程节点用独立表而非 JSONB** | 支持灵活查询和排序更新，行程模板和合同快照均按此方式存储 |
| **合同行程快照独立存储** | `contract_days` + `contract_nodes` 独立于行程模板，签合同后修改模板不影响历史合同 |
| **extra_data 用 JSONB** | 各节点类型（交通/住宿/景点/餐饮）的扩展字段差异大，JSONB 比每个字段加 nullable 列更整洁，PostgreSQL 对 JSONB 索引支持好 |
| **travelers 用 JSONB** | 出行人员是一个小的列表结构，数量不大，不值得单独建表 |
| **bills 统一一张表** | 入账/出账通过 `bill_type` 区分，共用支付方式、凭证等字段，便于汇总查询 |

## 四、财务汇总视图（可选）

```sql
-- 合同财务汇总视图，为合同详情页提速
CREATE VIEW v_contract_finance AS
SELECT
    c.id AS contract_id,
    c.total_amount,
    COALESCE(SUM(b.amount) FILTER (WHERE b.bill_type = 'income'), 0) AS total_received,
    c.total_amount - COALESCE(SUM(b.amount) FILTER (WHERE b.bill_type = 'income'), 0) AS pending_amount,
    COALESCE(SUM(b.amount) FILTER (WHERE b.bill_type = 'expense'), 0) AS total_expense,
    COALESCE(SUM(b.amount) FILTER (WHERE b.bill_type = 'income'), 0)
        - COALESCE(SUM(b.amount) FILTER (WHERE b.bill_type = 'expense'), 0) AS profit
FROM contracts c
LEFT JOIN bills b ON b.contract_id = c.id
GROUP BY c.id, c.total_amount;
```
